# 观察者模式（发布订阅模式/消息模式）

  1. 定义：定义一个主体（消息中心）：各种不同的消息。定义多个个体：订阅不同的消息，当主体发布某条消息，则会收到通知。
  2. 例子：
var msgCenter = (function () {
            var _msg = {} // 存储消息
            /*
               _msg = {
                    'carInfo': [p1.alreadyRegister.carInfo, p2.alreadyRegister.carInfo, p3.alreadyRegister.carInfo],
                    'newInfo': [p1.alreadyRegister.newInfo]
               } 
            */
            return {
                register: function (type, fn) { // 用于订阅一个消息
                    if (_msg[type]) {
                        _msg[type].push(fn)
                    } else {
                        _msg[type] = [fn]
                    }
                },
                fire: function (type, args) { // 用于发布消息
                    if (!_msg[type]) {
                        return
                    }
                    var event = {
                        type: type,
                        args: args || {}
                    }
                    for (var i = 0; i < _msg[type].length; i++) {
                        _msg[type][i](event);
                    }
                },
                cancel: function (type, fn) { // 用于取消订阅消息
                    if (!_msg[type]) {
                        return
                    }
                    for (var i = 0; i < _msg[type].length; i++) {
                        if (_msg[type][i] === fn) {
                            _msg[type][i].splice(i, 1)
                            break
                        }
                    }
                },
            }
        })()

        function Person() {
            this.alreadyRegister = {}
        }
        Person.prototype.register = function (type, fn) {
            if (this.alreadyRegister[type]) {
                console.log('已订阅,重复')
            } else {
                msgCenter.register(type, fn)
                this.alreadyRegister[type] = fn
            }
        }
        Person.prototype.cancel = function (type) {
            msgCenter.cancel(type, this.alreadyRegister[type])
            delete this.alreadyRegister[type]
        }

        var p1 = new Person()
        var p2 = new Person()
        var p3 = new Person()

        p1.register('carInfo', function (e) {
            console.log('p1' + e.type + '---' + e.args.info)
        })

        p1.register('newInfo', function (e) {
            console.log('p1' + e.type + '---' + e.args.info)
        })

        p2.register('carInfo', function (e) {
            console.log('p2' + e.type + '---' + e.args.info)
        })

        p3.register('carInfo', function (e) {
            console.log('p3' + e.type + '---' + e.args.info)
        })

        msgCenter.fire('carInfo', {
            info: '新款上市'
        })
        msgCenter.fire('newInfo', {
            info: '最新消息'
        })

        p1.register('carInfo', function (e) {
            console.log('p1' + e.type + '---' + e.args.info)
        })
